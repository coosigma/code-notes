TypeScript Playground

TypeScript / JavaScript Playground

This page lets you edit TypeScript or JavaScript and run it on demand. The code is executed only when you click the Run button.

import {Inputs} from "@observablehq/inputs"

// --- 1) Editable TS / JS source -------------------------------------------

DEFAULT_CODE = [
  "// You can write TypeScript or JavaScript here.",
  "// The code will run ONLY when you click the Run button.",
  "",
  "export function greet(name: string): string {",
  "  return `Hello, ${name}!`;",
  "}",
  "",
  "console.log(greet(\"World\"));"
].join("\n")

// Use a plain <textarea> so editing does NOT cause automatic re-runs.
editor = html`<textarea id="ts-editor"
  spellcheck="false"
  style="width:100%;min-height:220px;font-family:ui-monospace, SFMono-Regular, Menlo, Monaco, Consolas, 'Liberation Mono', 'Courier New', monospace;font-size:13px;line-height:1.5;border:1px solid #ddd;border-radius:6px;padding:8px;">${DEFAULT_CODE}</textarea>`

// --- 2) Lazy loader for the TypeScript compiler ---------------------------

// Only load the TypeScript compiler the first time the user clicks Run.
loadTypeScript = (url = "https://unpkg.com/typescript@5.4.5/lib/typescript.js") =>
  new Promise((resolve, reject) => {
    if (window.ts) return resolve(window.ts);
    const s = document.createElement("script");
    s.src = url;
    s.async = true;
    s.onload = () => resolve(window.ts);
    s.onerror = reject;
    document.head.appendChild(s);
  })

// --- 3) Run button (manual trigger) ---------------------------------------

viewof run = Inputs.button("\u25b6")

// --- 4) Result cell: re-runs ONLY when Run is clicked ---------------------

runResult = {
  // Depend on the Run button value, not the DOM element itself.
  const clicks = run

  // Before the first click, do NOT execute any user code.
  if (clicks == null || clicks === 0) {
    return html`<div style="min-height:3rem;border:1px solid #e5e7eb;border-radius:6px;padding:0.5rem;background-color:#fafafa;"></div>`
  }

  const textarea = document.getElementById("ts-editor")
  const code = textarea ? textarea.value : ""

  // Transpile TS to JS using the compiler (JS is also accepted).
  let js
  try {
    const ts = await loadTypeScript()
    const out = ts.transpileModule(code, {
      compilerOptions: {
        module: ts.ModuleKind.None,
        target: ts.ScriptTarget.ES2018,
        sourceMap: false
      }
    })
    js = out.outputText
  } catch (e) {
    return html`<div style="min-height:3rem;border:1px solid #e5e7eb;border-radius:6px;padding:0.5rem;background-color:#fafafa;">
      <pre style="margin:0;color:#b30000;white-space:pre-wrap;">[Transpile error]\n${String(e?.stack || e)}</pre>
    </div>`
  }

  // Execute JS in the current page and capture console.log output.
  const logs = []
  const originalLog = console.log
  console.log = (...args) => {
    logs.push(args.map(x => String(x)).join(" "))
    originalLog(...args)
  }

  let error = null
  try {
    const fn = new Function("exports", js)
    fn({})
  } catch (e) {
    error = e
  } finally {
    console.log = originalLog
  }

  if (error) {
    return html`<div style="min-height:3rem;border:1px solid #e5e7eb;border-radius:6px;padding:0.5rem;background-color:#fafafa;">
      <pre style="margin:0;color:#b30000;white-space:pre-wrap;">[Runtime error]\n${String(error?.stack || error)}</pre>
    </div>`
  }

  const text = logs.length ? logs.join("\n") : "(no console.log output)"
  return html`<div style="min-height:3rem;border:1px solid #e5e7eb;border-radius:6px;padding:0.5rem;background-color:#fafafa;">
    <pre style="margin:0;white-space:pre-wrap;">${text}</pre>
  </div>`
}

// --- 5) Layout: editor + Run button + output ------------------------------
html`<div style="display:grid;gap:0.75rem;">
  <div>
    <div style="font:600 14px/1.4 system-ui;margin-bottom:4px;">TypeScript / JavaScript (editable)</div>
    ${editor}
  </div>
  <div>${viewof run}</div>
  <div>
    <div style="font:600 14px/1.4 system-ui;margin-bottom:4px;">Output</div>
    ${runResult}
  </div>
</div>`
